## Overview

The `AdjacencyListWeightedGraph` class represents a weighted graph using an adjacency list. It extends `AdjacencyListGraph` and stores each vertex with a dictionary mapping adjacent vertices to edge weights, making it space-efficient for sparse weighted graphs.

## Constructor

```python
AdjacencyListWeightedGraph(directed=False, vertices=None)
```

<ParamField path="directed" type="bool" default="False">
  Whether the graph is directed
</ParamField>

<ParamField path="vertices" type="list[str]" default="None">
  List of vertex labels to initialize the graph with
</ParamField>

```python
from dsa.graph import AdjacencyListWeightedGraph

# Create empty weighted graph
g = AdjacencyListWeightedGraph()

# Create directed weighted graph
g = AdjacencyListWeightedGraph(directed=True)

# Create with initial vertices
g = AdjacencyListWeightedGraph(vertices=['A', 'B', 'C'])
```

```python Shortest Path (Dijkstra)
from dsa.graph import AdjacencyListWeightedGraph
import sys

# Create city road network
g = AdjacencyListWeightedGraph()

g.add_edge('Seattle', 'Portland', 173)
g.add_edge('Seattle', 'Vancouver', 140)
g.add_edge('Portland', 'Eugene', 110)
g.add_edge('Vancouver', 'Kelowna', 390)
g.add_edge('Eugene', 'Medford', 165)
g.add_edge('Kelowna', 'Calgary', 610)

def dijkstra(graph, start, end):
    distances = {v: sys.maxsize for v in graph.vertices()}
    distances[start] = 0
    previous = {v: None for v in graph.vertices()}
    unvisited = set(graph.vertices())
    
    while unvisited:
        # Find minimum distance vertex
        current = min(unvisited, key=lambda v: distances[v])
        
        if distances[current] == sys.maxsize:
            break
        
        unvisited.remove(current)
        
        if current == end:
            break
        
        # Update neighbor distances
        for neighbor, weight in graph.adjacent_items(current):
            if neighbor in unvisited:
                new_distance = distances[current] + weight
                if new_distance < distances[neighbor]:
                    distances[neighbor] = new_distance
                    previous[neighbor] = current
    
    # Reconstruct path
    path = []
    current = end
    while current is not None:
        path.insert(0, current)
        current = previous[current]
    
    return path, distances[end]

path, distance = dijkstra(g, 'Seattle', 'Medford')
print(f"Shortest route: {' -> '.join(path)}")
print(f"Total distance: {distance} miles")
```

```python Flight Routes (Directed Weighted)
from dsa.graph import AdjacencyListWeightedGraph

# Create flight network with costs
g = AdjacencyListWeightedGraph(directed=True)

# Add flights (one-way with different costs)
g.add_edge('NYC', 'LAX', 350)
g.add_edge('LAX', 'NYC', 380)
g.add_edge('NYC', 'ORD', 150)
g.add_edge('ORD', 'NYC', 160)
g.add_edge('ORD', 'LAX', 200)
g.add_edge('LAX', 'ORD', 220)
g.add_edge('LAX', 'SFO', 80)
g.add_edge('SFO', 'LAX', 85)
g.add_edge('SFO', 'ORD', 250)

def find_cheapest_routes_from(graph, city):
    """Find all direct flights from a city sorted by cost."""
    routes = []
    for destination, cost in graph.adjacent_items(city):
        routes.append((destination, cost))
    return sorted(routes, key=lambda x: x[1])

print("Cheapest flights from LAX:")
for dest, cost in find_cheapest_routes_from(g, 'LAX'):
    print(f"  {dest}: ${cost}")

# Find all routes with one layover
def find_one_stop_routes(graph, start, end):
    """Find all routes from start to end with exactly one stop."""
    routes = []
    for middle in graph.adjacents(start):
        if graph.has_edge(middle, end):
            cost = graph.get_weight(start, middle) + graph.get_weight(middle, end)
            routes.append((start, middle, end, cost))
    return sorted(routes, key=lambda x: x[3])

print("\nOne-stop routes NYC to LAX:")
for route in find_one_stop_routes(g, 'NYC', 'LAX'):
    start, middle, end, cost = route
    print(f"  {start} -> {middle} -> {end}: ${cost}")
```

```python Network Routing
from dsa.graph import AdjacencyListWeightedGraph
from collections import defaultdict

# Create network with latencies (ms)
g = AdjacencyListWeightedGraph()

g.add_edge('Router1', 'Router2', 5)
g.add_edge('Router1', 'Router3', 10)
g.add_edge('Router2', 'Router4', 8)
g.add_edge('Router3', 'Router4', 3)
g.add_edge('Router2', 'Router5', 15)
g.add_edge('Router4', 'Router5', 6)

# Find all paths with total latency
def find_all_paths_weighted(graph, start, end, path=None, weight=0):
    if path is None:
        path = []
    path = path + [start]
    
    if start == end:
        return [(path, weight)]
    
    paths = []
    for neighbor, edge_weight in graph.adjacent_items(start):
        if neighbor not in path:
            new_paths = find_all_paths_weighted(
                graph, neighbor, end, path, weight + edge_weight
            )
            paths.extend(new_paths)
    
    return paths

all_paths = find_all_paths_weighted(g, 'Router1', 'Router5')
print("All paths from Router1 to Router5:")
for path, latency in sorted(all_paths, key=lambda x: x[1]):
    print(f"  {' -> '.join(path)}: {latency}ms")

# Find k shortest paths
def k_shortest_paths(graph, start, end, k=3):
    all_paths = find_all_paths_weighted(graph, start, end)
    sorted_paths = sorted(all_paths, key=lambda x: x[1])
    return sorted_paths[:k]

print(f"\n{3} shortest paths:")
for i, (path, latency) in enumerate(k_shortest_paths(g, 'Router1', 'Router5', 3), 1):
    print(f"{i}. {' -> '.join(path)}: {latency}ms")
```

```python Minimum Spanning Tree (Prim's Algorithm)
from dsa.graph import AdjacencyListWeightedGraph
import sys

# Create graph
g = AdjacencyListWeightedGraph()

g.add_edge('A', 'B', 4)
g.add_edge('A', 'H', 8)
g.add_edge('B', 'C', 8)
g.add_edge('B', 'H', 11)
g.add_edge('C', 'D', 7)
g.add_edge('C', 'F', 4)
g.add_edge('C', 'I', 2)
g.add_edge('D', 'E', 9)
g.add_edge('D', 'F', 14)
g.add_edge('E', 'F', 10)
g.add_edge('F', 'G', 2)
g.add_edge('G', 'H', 1)
g.add_edge('G', 'I', 6)
g.add_edge('H', 'I', 7)

def prims_mst(graph):
    """Find minimum spanning tree using Prim's algorithm."""
    start = graph.vertices()[0]
    visited = {start}
    mst_edges = []
    total_weight = 0
    
    while len(visited) < len(graph.vertices()):
        min_edge = None
        min_weight = sys.maxsize
        
        # Find minimum weight edge connecting visited to unvisited
        for v in visited:
            for neighbor, weight in graph.adjacent_items(v):
                if neighbor not in visited and weight < min_weight:
                    min_weight = weight
                    min_edge = (v, neighbor, weight)
        
        if min_edge:
            start, end, weight = min_edge
            mst_edges.append(min_edge)
            total_weight += weight
            visited.add(end)
    
    return mst_edges, total_weight

mst, total = prims_mst(g)
print("Minimum Spanning Tree:")
for start, end, weight in mst:
    print(f"  {start} - {end}: {weight}")
print(f"Total weight: {total}")
```

## Performance

- **Space Complexity:** O(V + E) where V = vertices, E = edges
- **Add Vertex:** O(1)
- **Delete Vertex:** O(V + E)
- **Add Edge:** O(1) average case
- **Delete Edge:** O(1) average case (dictionary deletion)
- **Has Edge:** O(1) average case
- **Get Weight:** O(1) average case
- **Get Adjacents:** O(1)
- **Get Adjacent Items:** O(1)
- **Get All Edges:** O(V + E)

## Notes

- Best for sparse weighted graphs where E is much less than V²
- More space-efficient than weighted adjacency matrix
- Fast edge weight lookups using dictionary structure
- Weights can be any numeric type (int, float, etc.)
- Vertex labels must be strings
- Ideal for shortest path algorithms (Dijkstra, Bellman-Ford)
- Good for minimum spanning tree algorithms (Prim, Kruskal)
- For undirected graphs, edge weights are stored in both directions
- Adjacent items are stored as dictionary items for efficient iteration
